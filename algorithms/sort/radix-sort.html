const int K = 100;

int n, w[K], k, cnt[W];

struct Element {
  int key[K];

  bool operator<(const Element& y) const {
    // ä¸¤ä¸ªå…ƒç´ çš„æ¯”è¾ƒæµç¨‹

void counting_sort(int p) {
  memset(cnt, 0, sizeof(cnt));

void radix_sort(int n, int a[]) {
  int *b = new int[n];  // ä¸´æ—¶ç©ºé—´
  int *cnt = new int[1 << 8];
  int mask = (1 << 8) - 1;
  int *x = a, *y = b;
  for (int i = 0; i < 32; i += 8) {
    for (int j = 0; j != (1 << 8); ++j) cnt[j] = 0;
    for (int j = 0; j != n; ++j) ++cnt[x[j] >> i & mask];
    for (int sum = 0, j = 0; j != (1 << 8); ++j) {
      // ç­‰ä»·äº std::exclusive_scan(cnt, cnt + (1 << 8), cnt, 0);

    for (int j = 0; j != n; ++j) y[cnt[x[j] >> i & mask]++] = x[j];
    std::swap(x, y);
  }
  delete[] cnt;
  delete[] b;
}
